def problem8(L, N):  # L = long number, N = number of adjacent digits
    long_string = str(L)
    greatest = 0
    M = len(long_string) - N + 1
    for i in range(M):
        # get a chunk of the number
        short = long_string[i:i+N]

        # calculate the product
        product = 1
        for n in short:
            product *= int(n)

        # keep the largest value
        greatest = max(product, greatest)

    return greatest
